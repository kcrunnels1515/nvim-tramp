3. Installation                                      *nvim-tramp-installation*

Using lazy.nvim:
>lua
  {
    "username/nvim-tramp",
    dependencies = {
      "nvim-lua/plenary.nvim", -- if needed
    },
    config = function()
      require("nvim-tramp").setup({
        -- your configuration
      })
    end
  }
<

Using packer.nvim:
>lua
  use {
    'username/nvim-tramp',
    requires = { 'nvim-lua/plenary.nvim' }, -- if needed
    config = function()
      require('nvim-tramp').setup({
        -- your configuration
      })
    end
  }
<
